# コード　：　
from pathlib import Path
path = Path.cwd()     # 現在地を取得
print(path)

data_path = path/'data'  # dataフォルダ
print(data_path)

data_g = data_path.iterdir()
for x in data_g:             #dataフォルダの中から「ファイルだけ」を表示
    if x.is_file():
        print(x.name)

backup_path = data_path / 'backup'
backup_path.mkdir(exist_ok=True)      # data内にバックアップフォルダを作る
print(backup_path)

txt_files = data_path.glob('*.txt')
for x in txt_files:        # data内の.txtファイルだけ取得
    print(x.name)
