# コード
sales = {
    '姫路店': {'売上': 350000, '目標': 300000},
    '大阪店': {'売上': 280000, '目標': 300000},
    '神戸店': {'売上': 400000, '目標': 350000}
}

# 問題：
# 各店舗について、「達成」「未達成」を表示してください。
# 出力例：
# 姫路店 → 達成
# 大阪店 → 未達成
# 神戸店 → 達成

for name in sales:
    data = sales[name]
    if data['目標'] < data['売上']:
        print(f'{name}は達成')
    else:
        print(f'{name}は未達成')
