items = []

def add_item(items):
    print('---アイテム追加---')
    brand = input('ブランド名を入力してください：')
    name = input('アイテム名を入力してください：')
    price = int(input('値段を入力してください：'))
    note = input('メモ(なければ空でもOK)：')

    item = {
        'brand': brand,
        'name':name,
        'price':price,
        'is_bought':False,
        'note':note
    }
    items.append(item)
    print(f'{brand}/{name} を追加しました。')

def show_items(items):
    if not items:
        print('アイテムはまだありません。')
        return
    print('---アイテム一覧---')
    for item in items:
        status ='購入ずみ' if item['is_bought'] else '未購入'
        print(f"{item['brand']} / {item['name']} - {item['price']}円 [{status}] メモ:{item['note']}")
    
def show_total(items):
    total = 0
    for item in items:
        if not item['is_bought']:
            total += item['price']
    print(f'未購入アイテムの合計金額は{total}円です。')

def mark_as_bought(items):
    if not items:
        print('アイテムはまだありません。')
        return
    print('購入済みに変更するアイテムを選んでください。')
    for i,item in enumerate(items,start=1):
        status ='購入済み' if item['is_bought'] else '未購入'
        print(f"{i}.{item['brand']} / {item['name']} - {item['price']}円 [{status}]")

    num = int(input('購入済みに変更したい番号を入力してください。'))
    index = num - 1
    if 0 <= index < len(items):
        items[index]['is_bought'] = True
        print(f"{items[index]['brand']} / {items[index]['name']} を購入済みに変更しました")
    else:
        print('不正な番号です。')






def show_menu():
    print('===WishList メニュー===')
    print('1.アイテムを追加する')
    print('2.アイテム一覧表示')
    print('3.未購入アイテムの合計金額を表示する')
    print('4.購入済みに変更する')
    print('5.アプリを終了する')


while True:
    show_menu()
    choice = input('番号を選んでください。')
    if choice == '1':
        add_item(items)
    
    elif choice == '2':
        show_items(items)
    
    elif choice =='3':
        show_total(items)
    
    elif choice == '4':
        mark_as_bought(items)

    elif choice =='5':
        break

    else:
        print('無効な数字です。もう一度入力してください。')
