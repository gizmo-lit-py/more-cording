def show_users(users):
    for user in users:
        print(f'{user["name"]}さん{user["age"]}歳/{user["city"]}')

def search_users(users,keyword):
    found = False
    for user in users:
        if keyword in user['name']:
            print(f'{user["name"]}({user["age"]}歳)')
            found = True
    if not found:
            print('見つかりませんでした。')

def remove_user(users,keyword):
    found = False
    for user in users:
        if user['name'] == keyword:
            users.remove(user)
            print(f'{user["name"]}は削除しました')
            found = True
            break
    if not found:
        print('そのユーザーは見つかりませんでした')
