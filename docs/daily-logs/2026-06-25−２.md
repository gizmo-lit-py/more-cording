<?php

class Track
{
    public string $title;
    public int $bpm;
    public string $key;

    public function __construct(string $title, int $bpm, string $key)
    {
        $this->title = $title;
        $this->bpm = $bpm;
        $this->key = $key;
    }

    public function showInfo(): void
    {
        echo "曲名:" . $this->title . "\n";
        echo "BPM:" . $this->bpm . "\n";
        echo "キー:" . $this->key . "\n";
    }

    public function isBpmClose(int $targetBpm): bool
    {
        $difference = abs($this->bpm - $targetBpm);
        return $difference <= 5;
    }

    public function isKeyMatch(string $targetKey): bool
    {
        return $this->key === $targetKey;
    }
}

$tracks = [
    new Track("Tokyodrift", 100, "8A"),
    new Track("Track B", 124, "9A"),
    new Track("Track C", 132, "8A"),
    new Track("Track D", 128, "8A"),
];


$targetBpm = 128;
$targetKey = "8A";

foreach ($tracks as $track) {
    $track->showInfo();

    if ($track->isBpmClose($targetBpm)) {
        echo "BPMが近いです\n";
    } else {
        echo "BPMが離れています\n";
    }

    if ($track->isKeyMatch($targetKey)) {
        echo "キーが一致しています\n";
    } else {
        echo "キーが違います\n";
    }
}


function showRecommendedTracks(array $tracks, int $targetBpm, string $targetKey): void
{
    echo "おすすめ曲\n";
    echo "--------------------\n";

    foreach ($tracks as $track) {
        if ($track->isBpmClose($targetBpm) && $track->isKeyMatch($targetKey)) {
            $track->showInfo();
            echo "--------------------\n";
        }
    }
}


showRecommendedTracks($tracks, $targetBpm, $targetKey);
